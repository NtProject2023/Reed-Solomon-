def reverse_binary_search(arr, target):
    low = G(0)
    high = G(len(arr) - 1)

    while low <= high:
        mid = G((low + high) // 2)
        if arr[mid] < target:
            high = mid-1
        else:
            low = mid + 1
    
    if high >= G(0):
            return high
    else:
        return G(0)
